\section{Introduction}
\begin{frame}
\frametitle{What is Beamer and why use it?}

\begin{itemize}
\item A document class that creates presentations in \LaTeX
\pause\item The resulting .pdf will always appear the same
		\begin{itemize}
		\pause \item Easy to use on another computer
		\pause \item The content is never jumbled
		\end{itemize}
\pause\item It is easy to include mathematics
\pause\item It is pretty
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Disclaimers}

\begin{enumerate}
\item Errors and warnings are standard
		\visible<2->{\begin{itemize}
		\item Compile often
		\item Do not stress about getting rid of every warning
		\end{itemize}}
\visible<3->{\item Package conflicts may occur
		\begin{itemize}
		\visible<4->{\item Check the Beamer class documentation for details}
		\end{itemize}}
\visible<5->{\item It takes time to compile}
\end{enumerate}
\end{frame}

\section{Basic Commands}
\subsection{Structure}
\begin{frame}[fragile]
\frametitle{Frames}

\begin{itemize}
\item Every slide is made using the \texttt{frame} environment
\pause\item The title of slide is set by using \verb|\frametitle{TITLE}|
\end{itemize}

\pause\begin{example}
\verb|\begin{frame}|\\
\verb|\frametitle{Frames}|\\
\verb|CONTENT|\\
\verb|\end{frame}|
\end{example}
\end{frame}

\subsection{Organization}
\begin{frame}
\frametitle{Sections and Columns}

\begin{columns}
\begin{column}{.5\textwidth}
\begin{itemize}
\item The \texttt{section} and \texttt{subsection} commands can be used to organize the presentation
\visible<2->{\item Be sure to compile multiple times}
\end{itemize}
\end{column}

\begin{column}{.5\textwidth}
\begin{itemize}
\visible<3->{\item The \texttt{columns} environment can let you make multiple columns on your slide}
\visible<4->{\item Each column is made by the \texttt{column} environment {\em within} the \texttt{columns} environment}
\visible<5->{\item Each column has its own width}
\end{itemize} 
\end{column}
\end{columns}

\visible<6->{
\begin{block}{Note}
\begin{itemize}
\item It is possible to move back and forth between single and multiple columns
\item The positioning of columns can be specified
\end{itemize}
\end{block}}
\end{frame}

\subsection{Environments}
\begin{frame}
\frametitle{Blocks}

\begin{block}{Blocks}
A \texttt{block} is an environment that can be filled with any important information.\end{block}

\visible<2->{
\begin{block}{Types}
Some predefined blocks are \texttt{definition}, \texttt{theorem}, \texttt{proof}, \texttt{example}.
\end{block}}
\end{frame}

\subsection{Easter Egg!}
\begin{frame}
\frametitle{Overlay Specifications}

\begin{block}{}
As with any presentation software, effects can be added to a Beamer presentation
\end{block}

\only<10->{\begin{block}{}
Material does not have to appear linearly
\end{block}}

\begin{columns}[t]
\visible<2->{\begin{column}{.5\textwidth}
\[\textbf{What are the options?}\]

\begin{itemize}
\item \texttt{pause}\only<2>{: content between pauses appears when the slide advances}
\item<3-> \texttt{(un)cover}\only<3>{: the content is transparent except for the specified frames}
\item<4-> \texttt{(in)visible}\only<4>{: the content is always ``present'' but only (in)visible where specified}
\item<5-> \texttt{only}\only<5>{: the content is only ``present'' on the specified frames}
\end{itemize}
\end{column}}

\visible<6->{\begin{column}{.5\textwidth}
\[\textbf{To what do these apply?}\]

\begin{itemize}
\item \textbf<7>{Font effects} (bold, emphasis, etc.)
\item \textcolor<8>{blue}{\textbf<8>{Colors}}
\item \alert<9>{``Alerts''}
\item List environments
\item Entries of a table
\item Lines of an equation
\end{itemize}
\end{column}}
\end{columns}
